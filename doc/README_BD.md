# TABELAS
### TABELA ADMINISTRADOR
```sql
CREATE TABLE ADMINISTRADOR (
    LOGIN VARCHAR(20),
    SENHA VARCHAR(50)
);
```

### TABELA USUARIOS

```sql
CREATE TABLE USUARIOS (
    CPF VARCHAR(11) PRIMARY KEY NOT NULL,
    NOME varchar(100),
    SOBRENOME varchar(100),
    EMAIL VARCHAR(100),
    TELEFONE INT(13),
    DATA_NASCIMENTO DATE
)
```

### TABELA LIVROS
```sql
create table LIVROS (
    ISBN VARCHAR(13) PRIMARY KEY NOT NULL,
    AUTOR varchar(50),
    TITULO VARCHAR(100) NOT NULL,
    DESCRICAO TINYTEXT,
    GENERO varchar(20),
    UNIDADES_DISPONIVEIS int not null,
    EDITORA varchar(50),
    ANO_PUBLICACAO DATE
)
```

### TABELA EMPRESTIMOS
```sql
CREATE TABLE EMPRESTIMO(
    ID INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    LIVRO_ISBN VARCHAR(13) NOT NULL,
    CPF_PESSOA VARCHAR(11) NOT NULL,
    DATA_EMPRESTADO DATE NOT NULL,
    TEMPO_EMPRESTIMO INT NOT NULL,
    STATUS_LIVRO VARCHAR(15) NOT NULL,
    FOREIGN KEY (CPF_PESSOA) REFERENCES USUARIOS(CPF),
    FOREIGN KEY (LIVRO_ISBN) REFERENCES LIVROS(ISBN)
)
```
